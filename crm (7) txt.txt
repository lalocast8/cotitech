function rActs(){
  // Combinar actividades manuales + seguimientos de tratos (como actividades virtuales)
  const today=td0();

  // Generar actividades desde tratos con followUpDate que no tengan actividad pendiente ligada
  const dealActs=deals.filter(d=>d.followUpDate&&d.status==='open').map(d=>{
    // Ver si ya existe una actividad pendiente ligada a este trato
    const existing=activities.find(a=>a.dealId===d.id&&!a.done);
    if(existing)return null; // ya existe, no duplicar
    // Crear virtual activity desde el trato
    return{
      id:'deal_'+d.id,
      type:d.followUpType||'call',
      title:(AL[d.followUpType||'call']||'Contacto con')+' '+d.company,
      desc:d.followUpDesc||'Seguimiento programado',
      contactName:d.company,
      date:d.followUpDate,
      time:'📅 '+new Date(d.followUpDate+'T00:00:00').toLocaleDateString('es-MX',{day:'2-digit',month:'short',year:'numeric'}),
      dealId:d.id,
      done:false,
      doneAt:null,
      virtual:true
    };
  }).filter(Boolean);

  // Unir todas las actividades
  const allActs=[...activities,...dealActs].sort((a,b)=>{
    const da=a.date||'9999',db=b.date||'9999';
    return da.localeCompare(db);
  });

  // Filtros
  const filterAct=(a)=>{
    if(a.done)return actTab==='done';
    if(actTab==='done')return false;
    if(actTab==='pending')return true;
    if(actTab==='overdue'){const d=new Date(a.date+'T00:00:00');return a.date&&d<today;}
    if(actTab==='today'){const d=new Date(a.date+'T00:00:00');return a.date&&Math.round((d-today)/86400000)===0;}
    if(actTab==='week'){const d=new Date(a.date+'T00:00:00');const diff=Math.round((d-today)/86400000);return a.date&&diff>=0&&diff<=7;}
    if(actTab==='month'){const d=new Date(a.date+'T00:00:00');const diff=Math.round((d-today)/86400000);return a.date&&diff>=0&&diff<=30;}
    return true;
  };
  const filt=allActs.filter(filterAct);

  const counts={
    pending:allActs.filter(a=>!a.done).length,
    done:allActs.filter(a=>a.done).length,
    overdue:allActs.filter(a=>!a.done&&a.date&&new Date(a.date+'T00:00:00')<today).length,
    today:allActs.filter(a=>!a.done&&a.date&&Math.round((new Date(a.date+'T00:00:00')-today)/86400000)===0).length,
    week:allActs.filter(a=>!a.done&&a.date&&Math.round((new Date(a.date+'T00:00:00')-today)/86400000)>=0&&Math.round((new Date(a.date+'T00:00:00')-today)/86400000)<=7).length,
    month:allActs.filter(a=>!a.done&&a.date&&Math.round((new Date(a.date+'T00:00:00')-today)/86400000)>=0&&Math.round((new Date(a.date+'T00:00:00')-today)/86400000)<=30).length,
  };

  const tabs=[
    {k:'pending',label:'⏳ Pendientes',n:counts.pending},
    {k:'overdue',label:'🔴 Vencidas',n:counts.overdue},
    {k:'today',label:'🟠 Hoy',n:counts.today},
    {k:'week',label:'📅 7 días',n:counts.week},
    {k:'month',label:'📆 30 días',n:counts.month},
    {k:'done',label:'✅ Completadas',n:counts.done},
  ];

  return`<div class="page-header"><div><p class="page-title">Actividades</p><p class="page-subtitle">${counts.pending} pendientes · ${counts.overdue} vencidas · ${counts.done} completadas</p></div><button class="btn btn-accent" onclick="openNewActivity()">+ Nueva actividad</button></div>
  <div style="display:flex;gap:2px;background:var(--surface2);border-radius:var(--rad);padding:3px;margin-bottom:16px;flex-wrap:wrap">
    ${tabs.map(t=>`<div class="tab ${actTab===t.k?'active':''}" onclick="actTab='${t.k}';render()" style="white-space:nowrap">${t.label}${t.n>0?' ('+t.n+')':''}</div>`).join('')}
  </div>
  <div class="card"><div class="card-pad">
  ${filt.length?filt.map(a=>{
    const deal=a.dealId?deals.find(d=>d.id===a.dealId):null;
    const aDate=a.date?new Date(a.date+'T00:00:00'):null;
    const diff=aDate?Math.round((aDate-today)/86400000):null;
    const isOverdue=!a.done&&aDate&&aDate<today;
    const isToday=!a.done&&diff===0;
    const urgColor=isOverdue?'#fcebeb':isToday?'#fffbf3':'';
    return`<div class="act-item" style="border-radius:var(--rad);padding:10px 8px;${urgColor?'background:'+urgColor+';margin-bottom:4px;border:1px solid '+(isOverdue?'#f09595':'#fac775')+';':''}${a.done?'opacity:.55':''}">
      <div style="display:flex;flex-direction:column;align-items:center;gap:5px;flex-shrink:0;width:48px">
        <div class="act-icon ${a.type}" style="margin:0;${a.done?'filter:grayscale(.8)':''}">${AI[a.type]||'📌'}</div>
        ${!a.done
          ?`<button onclick="completeActById('${a.id}',${a.virtual?true:false})" style="background:none;border:1px solid var(--accent);border-radius:999px;cursor:pointer;font-size:9px;padding:2px 5px;color:var(--accent);font-weight:600;white-space:nowrap">✓ Listo</button>`
          :`<span style="font-size:9px;color:var(--accent);font-weight:600">✅</span>`
        }
      </div>
      <div style="flex:1;min-width:0">
        <div style="font-size:13px;font-weight:500;${a.done?'text-decoration:line-through;color:var(--text3)':''}">
          ${isOverdue?'🔴 ':''}${isToday?'🟠 ':''}${esc(a.title)}
          ${a.virtual?'<span style="font-size:10px;background:#e6f1fb;color:#185fa5;border-radius:3px;padding:1px 5px;margin-left:4px">Pipeline</span>':''}
        </div>
        <div style="font-size:12px;color:var(--text3);margin-top:2px">${esc(a.desc)}</div>
        ${deal?`<div style="font-size:11px;color:var(--text2);margin-top:3px;display:flex;align-items:center;gap:6px">
          <span>🏢 <strong>${esc(deal.name)}</strong> · ${esc(deal.company)}</span>
          ${deal.seller?`<span class="badge bg-gray" style="font-size:10px">👤 ${esc(deal.seller)}</span>`:''}
          ${deal.periodicity?`<span style="font-size:10px;color:var(--text3)">🔄 cada ${deal.periodicity} días</span>`:''}
        </div>`:''}
        <div style="font-size:11px;color:${isOverdue?'#a32d2d':isToday?'#854f0b':'var(--text3)'};margin-top:3px;font-weight:${isOverdue||isToday?600:400}">
          ${a.date?new Date(a.date+'T00:00:00').toLocaleDateString('es-MX',{weekday:'short',day:'2-digit',month:'short',year:'numeric'}):'Sin fecha'}
          ${isOverdue?' · ⚠️ Vencida hace '+Math.abs(diff)+' día'+(Math.abs(diff)===1?'':'s'):''}
          ${isToday?' · ⚡ Hoy':''}
          ${!isOverdue&&!isToday&&diff!==null&&diff>0?' · En '+diff+' día'+(diff===1?'':'s'):''}
          ${a.done&&a.doneAt?' · ✅ Completada: '+a.doneAt:''}
        </div>
      </div>
      ${!a.virtual?`<button class="btn btn-sm btn-danger" onclick="deleteActivity('${a.id}')" style="align-self:flex-start;flex-shrink:0">🗑️</button>`:''}
    </div>`;
  }).join(''):`<div class="empty"><div class="empty-icon">${actTab==='done'?'✅':'📭'}</div><p>Sin actividades en esta categoría</p></div>`}
  </div></div>`;
}
function completeActById(rawId, isVirtual){
  if(isVirtual){
    const dealId=parseInt(String(rawId).replace('deal_',''));
    const deal=deals.find(d=>d.id===dealId);
    if(!deal)return;
    const now=new Date();
    // Guardar la actividad como completada
    const completedAct={
      id:nid(activities),
      type:deal.followUpType||'call',
      title:(AL[deal.followUpType||'call']||'Contacto con')+' '+deal.company,
      desc:deal.followUpDesc||'Seguimiento',
      contactName:deal.company,
      date:deal.followUpDate,
      time:now.toLocaleDateString('es-MX',{day:'2-digit',month:'short'}),
      dealId:dealId,
      done:true,
      doneAt:now.toLocaleDateString('es-MX',{day:'2-digit',month:'short',year:'numeric'}),
      virtual:false
    };
    activities.push(completedAct);
    // Limpiar followUpDate del trato para que desaparezca la actividad virtual
    const oldType=deal.followUpType;
    const oldDesc=deal.followUpDesc;
    deal.followUpDate=null;
    deal.followUpDesc='';
    deal.followUpType='';
    saveAll();
    render(); // actualiza la lista inmediatamente antes del modal
    scheduleNext(deal,oldType||'call',oldDesc||'Seguimiento',completedAct.id);
  } else {
    const id=typeof rawId==='string'?parseInt(rawId):rawId;
    completeActivity(id);
  }
}

}
function openNewActivity(){
  document.getElementById('mt').textContent='Nueva actividad';
  const mb=document.getElementById('mb');
  mb.innerHTML=`
    <div class="form-row">
      <div class="form-field"><label>Tipo</label>
        <select id="at"><option value="call">📞 Llamada</option><option value="email">✉️ Email</option><option value="meet">📅 Reunión</option><option value="note">📝 Nota</option></select>
      </div>
      <div class="form-field"><label>Fecha</label>
        <input type="date" id="adate" value="${new Date().toISOString().split('T')[0]}">
      </div>
    </div>
    <div class="form-field"><label>Ligar a trato (opcional)</label>
      <select id="adeal">
        <option value="">Sin trato asociado</option>
        ${deals.filter(d=>d.status==='open').map(d=>`<option value="${d.id}">${esc(d.name)} · ${esc(d.company)}</option>`).join('')}
      </select>
    </div>
    <div class="form-field"><label>Descripción *</label>
      <textarea id="ad" placeholder="¿Qué pasó o qué se acordó?"></textarea>
    </div>`;
  mFn=()=>{
    const desc=mb.querySelector('#ad').value.trim();
    if(!desc){toast('Descripción requerida');return;}
    const t=mb.querySelector('#at').value;
    const dateVal=mb.querySelector('#adate').value;
    const dealIdVal=mb.querySelector('#adeal').value;
    const dealId=dealIdVal?parseInt(dealIdVal):null;
    const deal=dealId?deals.find(d=>d.id===dealId):null;
    const cn=deal?deal.company:'';
    const now=new Date();
    activities.push({id:nid(activities),type:t,title:AL[t]+' '+(cn||''),desc,contactName:cn,date:dateVal||now.toISOString().split('T')[0],time:now.toLocaleDateString('es-MX',{day:'2-digit',month:'short'})+' '+now.toLocaleTimeString('es-MX',{hour:'2-digit',minute:'2-digit'}),dealId,done:false,doneAt:null,virtual:false});
    // Si tiene trato y fecha, actualizar followUp del trato también
    if(deal&&dateVal){deal.followUpDate=dateVal;deal.followUpDesc=desc;deal.followUpType=t;}
    saveAll();closeModal();render();toast('Actividad registrada');
  };
  openModal();
}
function completeActivity(id){
  const a=activities.find(x=>x.id===id);if(!a)return;
  a.done=true;
  a.doneAt=new Date().toLocaleDateString('es-MX',{day:'2-digit',month:'short',year:'numeric'});
  const deal=a.dealId?deals.find(d=>d.id===a.dealId):null;
  const oldType=a.type,oldDesc=a.desc;
  // Limpiar followUpDate del trato para que no regenere la actividad virtual
  if(deal){deal.followUpDate=null;deal.followUpDesc='';deal.followUpType='';}
  saveAll();
  render(); // actualiza la lista inmediatamente antes del modal
  if(deal){scheduleNext(deal,oldType,oldDesc,a.id);}
  else{toast('✅ Actividad completada');}
}
function deleteActivity(rawId){
  const id=typeof rawId==='string'&&rawId.startsWith('deal_')?null:parseInt(rawId);
  if(!id){toast('Las actividades del pipeline se eliminan desde el trato');return;}
  activities=activities.filter(a=>a.id!==id);saveAll();render();toast('Actividad eliminada');
}
